2WordPress Plugin Integration

Enqueue the styles in your plugin:

PHP WordPress Enqueue 
function enqueue_scwp_admin_ui() {
    wp_enqueue_style(
        'scwp-admin-ui',
        plugin_dir_url(__FILE__) . 'assets/css/scwp-admin-ui.css',
        array(),
        '1.0.0'
    );
}
add_action('admin_enqueue_scripts', 'enqueue_scwp_admin_ui');

Usage Instructions & PHP Integration

Complete guide for implementing the Interactive Position Selector in WordPress plugins.

Basic JavaScript Interaction
JavaScript Position Selector Event Handling 
// Initialize Position Selector
document.addEventListener('DOMContentLoaded', function() {
    const positionOptions = document.querySelectorAll('.scwp-position-selector__option');
    const selectedDisplay = document.getElementById('selected-position');
    
    positionOptions.forEach(option => {
        option.addEventListener('click', function() {
            // Remove selected class from all options
            positionOptions.forEach(opt => opt.classList.remove('selected'));
            
            // Add selected class to clicked option
            this.classList.add('selected');
            
            // Update display text
            const position = this.getAttribute('data-position');
            selectedDisplay.textContent = formatPositionText(position);
            
            // Trigger custom event for form integration
            const event = new CustomEvent('positionChanged', {
                detail: { position: position }
            });
            document.dispatchEvent(event);
        });
    });
});

function formatPositionText(position) {
    const positions = {
        'top-left': 'Top Left',
        'top-right': 'Top Right', 
        'bottom-left': 'Bottom Left',
        'bottom-right': 'Bottom Right'
    };
    return positions[position] || position;
}

            <div class="scwp-position-selector__screen">

                         data-position="<?php echo esc_attr($position_key); ?>">
                        <span class="sr-only"><?php echo esc_html($position_label); ?></span>
                    </div>
                <?php endforeach; ?>
            </div>
            
            <!-- Hidden input to store selected value -->
            <input type="hidden" 
                   id="<?php echo esc_attr($field_name); ?>"

                   value="<?php echo esc_attr($current_value); ?>" />
        </div>
        
        <p class="description">

/**
 * Display Cookie Banner with Position from Settings
 */
function display_cookie_banner() {
    if (is_admin() || has_user_consent()) {
        return; // Don't show if admin or user already consented
    }
    
    $position = get_option('my_plugin_banner_position', 'top-right');
    $banner_text = get_option('my_plugin_banner_text', __('We use cookies to improve your experience.', 'textdomain'));
    
    // Position-specific CSS classes
    $position_classes = array(
        'top-left' => 'scwp-cookie-banner--top-left',
        'top-right' => 'scwp-cookie-banner--top-right',
        'bottom-left' => 'scwp-cookie-banner--bottom-left',
        'bottom-right' => 'scwp-cookie-banner--bottom-right'
    );
    
    $position_class = isset($position_classes[$position]) ? $position_classes[$position] : $position_classes['top-right'];
    ?>
    <div id="cookie-banner" class="scwp-cookie-banner <?php echo esc_attr($position_class); ?>">
        <div class="scwp-cookie-banner__content">
            <p class="scwp-cookie-banner__message">
                <?php echo esc_html($banner_text); ?>
            </p>
            
            <div class="scwp-cookie-banner__actions">
                <button type="button" 
                        class="scwp-btn scwp-btn--primary" 
                        onclick="acceptCookies()">
                    <?php _e('Accept', 'textdomain'); ?>
                </button>
                
                <button type="button" 
                        class="scwp-btn scwp-btn--secondary scwp-ml-2" 
                        onclick="showCookieSettings()">
                    <?php _e('Settings', 'textdomain'); ?>
                </button>
            </div>
        </div>
    </div>
    
    <script>
    function acceptCookies() {
        // Set cookie consent
        document.cookie = "cookie_consent=accepted; path=/; max-age=31536000";
        
        // Hide banner with animation
        const banner = document.getElementById('cookie-banner');
        banner.style.transform = 'translateY(-100%)';
        banner.style.opacity = '0';
        setTimeout(() => banner.remove(), 300);
        
        // Trigger custom event
        document.dispatchEvent(new CustomEvent('cookiesAccepted'));
    }
    
    function showCookieSettings() {
        // Open settings modal or redirect to settings page
        window.location.href = '<?php echo esc_url(admin_url('options-general.php?page=cookie-settings')); ?>';
    }
    </script>
    <?php
}

PHP AJAX Handler for Live Preview 
<?php
/**
 * AJAX handler for position preview
 */
function ajax_update_banner_position() {
    check_ajax_referer('banner_position_nonce');
    
    if (!current_user_can('manage_options')) {
        wp_die(__('Insufficient permissions', 'textdomain'));
    }
    
    $position = sanitize_text_field($_POST['position']);
    $valid_positions = array('top-left', 'top-right', 'bottom-left', 'bottom-right');
    
    if (!in_array($position, $valid_positions)) {
        wp_send_json_error(__('Invalid position', 'textdomain'));
    }
    
    // Generate preview HTML
    ob_start();
    ?>
    <div class="scwp-cookie-banner scwp-cookie-banner--<?php echo esc_attr($position); ?> preview-banner">
        <div class="scwp-cookie-banner__content">
            <p class="scwp-cookie-banner__message">
                <?php _e('Preview: We use cookies to improve your experience.', 'textdomain'); ?>
            </p>
            <div class="scwp-cookie-banner__actions">
                <button class="scwp-btn scwp-btn--primary"><?php _e('Accept', 'textdomain'); ?></button>
                <button class="scwp-btn scwp-btn--secondary"><?php _e('Settings', 'textdomain'); ?></button>
            </div>
        </div>
    </div>
    <?php
    $preview_html = ob_get_clean();
    
    wp_send_json_success(array(
        'position' => $position,
        'preview_html' => $preview_html
    ));
}
add_action('wp_ajax_update_banner_position', 'ajax_update_banner_position');
